import os
import pytest
import importlib

os.environ["AOC_TESTING"] = "TRUE"
solution = importlib.import_module("solutions.{year}.{day:02d}.aoc_{day:02d}_{year}")


@pytest.mark.parametrize("input_str, part1, part2", solution.TEST_CASES)
def test_part1(input_str, part1, part2):
    if part1 is None:
        pytest.skip("No test case for part 1")

    assert solution.part1(input_str) == part1


@pytest.mark.parametrize("input_str, part1, part2", solution.TEST_CASES)
def test_part2(input_str, part1, part2):
    if part2 is None:
        pytest.skip("No test case for part 2")

    assert solution.part2(input_str) == part2
