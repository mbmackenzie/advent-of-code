import pytest
import importlib

solution = importlib.import_module("{year}.day{day:02d}")


@pytest.mark.parametrize("input_str, part1, part2", solution.TEST_CASES)
def test_part1(input_str, part1, part2):
    if part1 is None:
        pytest.skip("No test case for part 1")

    assert solution.part1(input_str, testing=True) == part1


@pytest.mark.parametrize("input_str, part1, part2", solution.TEST_CASES)
def test_part2(input_str, part1, part2):
    if part2 is None:
        pytest.skip("No test case for part 2")

    assert solution.part2(input_str, testing=True) == part2
